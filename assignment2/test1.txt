
[* example of area of rectangle *]

$

function areaRectangle (integer length, integer width)
{
   return length *width;

}

    $
    [* declarations *]
    integer l , w;
    boolean valid;
    $
    
    scan(l,w);
    if (l * w < 0)
    {
        valid = False;
        print(valid);
    }

    else{
        print(areaRectangle(l,w));
    }
    endif
